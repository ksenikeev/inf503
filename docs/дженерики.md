//WildCard

public static void printList(List<?> list) {
    for (Object elem: list) {
        System.out.print(elem + " ");
    }
    System.out.println();
}

public static void printCatList(List<? extends Cat> list) {
    for (Cat cat: list) {
        System.out.print(cat + " ");
    }
    System.out.println();
}
